#include<stdio.h>
int main(int argc,char *argv[])
{
	FILE *fp=NULL;
	int iflag;

	if(argc <= 1)
	{
		printf("check usage of %s \n",argv[0]);
		return -1;
	}
	fp=fopen(argv[1],"w");
	if(fp==NULL)
	{
		printf("Open file %s failed!",argv[1]);
	}
	printf("this work is finished!");
	iflag=fclose(fp);
	if(iflag==0)
	{
		printf("Close file %s succeed!\n ",argv[1]);
		return 0;
	}
	else 
	{
		printf("Close file %s failed!",argv[1]);
		return 1;
	}
	
}
