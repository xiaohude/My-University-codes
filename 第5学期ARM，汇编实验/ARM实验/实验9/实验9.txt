#include "2440addr.h"
#include "2440lib.h"


void __irq adc_isr(void);

//adc初始化函数
void adc_init()
{
	//ADC参数初始化
	rADCCON &= ~((1<<14) | (0xff<<6) | (7<<3) | (1<<2) | (1<<1) | (1<<0) );
	rADCCON |= ((1<<14) | (49<<6));
	rADCTSC |= ~(1<<2);
	//清除中断相关寄存器
	rSUBSRCPND |= (1<<10);
	rSRCPND |= (1<<31);
	rINTPND |= (1<<31);
	//选择中断屏蔽
	rINTSUBMSK &= ~(1<<10);
	rINTMSK &= ~(1<<31);
	//安装中断处理函数
	pISR_ADC = (unsigned int)adc_isr;
}

volatile unsigned short ad_value0 = 0;
volatile unsigned short ad_value1 = 0;
volatile unsigned char adc_done = 0;
void __irq adc_isr(void)
{
	if( 0 != (rSUBSRCPND & (1<<10)) )
	{
		ad_value1 = (unsigned short)(rADCDAT0 & 0x3ff);
		adc_done = 1;
		rSUBSRCPND |= (1<<10);
	}
	rSRCPND |= (1<<31);
	rINTPND |= (1<<31);
}


void Main(void)
{	
	
	//初始化串口0以方便调试
	Uart0_Init_all();
	//调用初始化程序
	adc_init();
	Uart_Printf("\n%s\n", "开始adc中断程序，请大家仔细阅读程序~");
	while(1)
	{
		Uart_Printf("\n%s\n", "按下任意键，查询AIN0中和断读取AIN1~");
		Uart_Getch();
		
		rADCCON &= ~(7<<3);
		rADCCON |= ((0<<3) | (1<<1));
		ad_value0 = (unsigned short)(rADCDAT0 & 0x3ff);
		//等待ADC完成
		while(!(rADCCON & (1<<15)));
		//读取AD转换结果
		ad_value0 = (unsigned short)(rADCDAT0 & 0x3ff);
		Uart_Printf("\n%s%u\n", "查询读取的AIN0值为;", ad_value0);
		
		
		//选择adc通道为AIN1并启动
		adc_done = 0;
		rADCCON &= ~(7<<3);
		rADCCON |= ((1<<3) | (1<<0));
		//等待ADC完成
		while(rADCCON & (1<<0));//用来检验转换是否启动，不写也可以
		while(!adc_done);
		//AD转换结果
		Uart_Printf("\n%s%u\n", "中断读取的AIN1值为;", ad_value1);
	}
}
